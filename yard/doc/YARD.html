def self.windows?
  return @windows if defined? @windows
  require 'rbconfig'
  if ::RbConfig::CONFIG['host_os'] =~ /mingw|win32|cygwin/
    @windows = true
  else
    @windows = false
  end
ensure
  @windows ||= false
end
