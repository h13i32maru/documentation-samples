def parse(file)
  case @@gettext_version
  when 2
    parser = GetText::PoParser.new
    data = GetText::MoFile.new
  when 3
    parser = GetText::POParser.new
    data = GetText::MO.new
  end
  parser.report_warning = false
  parser.parse_file(file, data)
  data
end
