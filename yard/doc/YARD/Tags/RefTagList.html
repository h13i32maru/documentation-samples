def tags
  if owner.is_a?(CodeObjects::Base)
    o = owner.tags(tag_name)
    o = o.select {|t| t.name.to_s == name.to_s } if name
    o.each do |t|
      t.extend(RefTag)
      t.owner = owner
    end
    o
  else
    []
  end
end
