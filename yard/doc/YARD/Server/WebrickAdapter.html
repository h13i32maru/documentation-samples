def start
  server_options[:ServerType] = WEBrick::Daemon if server_options[:daemonize]
  server = WEBrick::HTTPServer.new(server_options)
  server.mount('/', WebrickServlet, self)
  trap("INT") { server.shutdown }
  server.start
end
